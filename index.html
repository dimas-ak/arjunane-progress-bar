-=[ Progress-bar.js DEMO ]=-

Dependency-free JavaScript library Progress Bar

Required:

  • jQuery.js
  • arjunane-pb.css
  • arjunane-pb.js

TYPE 1

                        
$('.strong').pb({
    percent: 30,
    title:'Judul apa saja',
    info: 'Info apa aja kang wkwkwk mencoba panjang saja wkwkwk'
});

TYPE 2

                        
$('#mencoba').pb({
    percent: 10,
    information: false,
    type: 2,
    color: '',
});

Update with Click

$('#mencoba').on('click', function(){
    $(this).pb_update({
        percent: 80,
        type: 2,
        information:true,
        title:'Judul apa saja',
        info: 'Info apa aja kang wkwkwk mencoba panjang saja wkwkwk',
        color:'red'
    });
});
